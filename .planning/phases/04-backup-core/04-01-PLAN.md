    /// Creates directory if it doesn't exist.
    fn backup_dir(&self) -> Result<std::path::PathBuf> {
        // We don't have db_path stored — read from env or use default
        let home = dirs::home_dir()
            .ok_or_else(|| EngramError::Database("could not determine home directory".into()))?;
        let dir = home.join(".engram").join("backups");
        std::fs::create_dir_all(&dir)
            .map_err(|e| EngramError::Database(format!("failed to create backup dir: {e}")))?;
        Ok(dir)
    }

    /// Query schema version from _migrations table.
    fn schema_version(&self) -> i32 {
        let conn = self.conn();
        conn.query_row("SELECT MAX(version) FROM _migrations", [], |row| {
            row.get::<_, Option<i32>>(0).map(|v| v.unwrap_or(0))
        })
        .unwrap_or(0)
    }

    /// Compute SHA-256 hex digest of a file.
    fn sha256_file(path: &std::path::Path) -> Result<String> {
        let mut file = std::fs::File::open(path)
            .map_err(|e| EngramError::Database(format!("failed to open file for checksum: {e}")))?;
        let mut hasher = Sha256::new();
        std::io::copy(&mut file, &mut hasher)
            .map_err(|e| EngramError::Database(format!("failed to compute checksum: {e}")))?;
        Ok(hasher.finalize().iter().map(|b| format!("{:02x}", b)).collect())
    }

```rust
    fn backup_create(&self, trigger: &str, label: Option<&str>) -> Result<BackupRecord> {
        let backup_dir = self.backup_dir()?;
        let now = Utc::now();
        // ISO 8601 with colons replaced by - for filesystem safety (D-02)
        let timestamp = now.format("%Y-%m-%dT%H-%M-%SZ").to_string();
        let backup_filename = format!("engram-{timestamp}.db");
        let backup_path = backup_dir.join(&backup_filename);

        // Use rusqlite::backup::Backup::run_to_completion() (BACKUP-12)
        {
            let conn = self.conn();
            let mut dst = rusqlite::Connection::open(&backup_path)
                .map_err(|e| EngramError::Database(format!("failed to create backup file: {e}")))?;
            let backup = rusqlite::backup::Backup::new(&conn, &mut dst)
                .map_err(|e| EngramError::Database(format!("failed to init backup: {e}")))?;
            backup
                .run_to_completion(500, Duration::from_millis(0), None)
                .map_err(|e| EngramError::Database(format!("backup failed: {e}")))?;
        }

        // File size
        let metadata = std::fs::metadata(&backup_path)
            .map_err(|e| EngramError::Database(format!("failed to read backup metadata: {e}")))?;
        let size_bytes = metadata.len();

        // SHA-256 checksum
        let sha256 = Self::sha256_file(&backup_path)?;

        // Stats: count observations, sessions, edges
        let conn = self.conn();
        let observations: usize = conn
            .query_row("SELECT COUNT(*) FROM observations", [], |row| {
                row.get::<_, i64>(0).map(|v| v as usize)
            })
            .unwrap_or(0);
        let sessions: usize = conn
            .query_row("SELECT COUNT(*) FROM sessions", [], |row| {
                row.get::<_, i64>(0).map(|v| v as usize)
            })
            .unwrap_or(0);
        let edges: usize = conn
            .query_row("SELECT COUNT(*) FROM edges", [], |row| {
                row.get::<_, i64>(0).map(|v| v as usize)
            })
            .unwrap_or(0);

        let stats = BackupStats {
            observations,
            sessions,
            edges,
        };

        // Schema version
        let schema_version = self.schema_version();

        // Write .meta.json sidecar (BACKUP-13)
        let meta = serde_json::json!({
            "version": env!("CARGO_PKG_VERSION"),
            "schema_version": schema_version,
            "created_at": now.to_rfc3339(),
            "trigger": trigger,
            "label": label,
            "size_bytes": size_bytes,
            "sha256": sha256,
            "stats": {
                "observations": stats.observations,
                "sessions": stats.sessions,
                "edges": stats.edges,
            }
        });
        let meta_path = backup_dir.join(format!("engram-{timestamp}.meta.json"));
        std::fs::write(&meta_path, serde_json::to_string_pretty(&meta)?)
            .map_err(|e| EngramError::Database(format!("failed to write meta.json: {e}")))?;

        Ok(BackupRecord {
            path: backup_path,
            created_at: now,
            trigger: trigger.to_string(),
            label: label.map(String::from),
            size_bytes,
            sha256,
            stats,
        })
    }

    fn backup_verify(&self, path: &std::path::Path) -> Result<BackupVerifyResult> {
        // Check file exists
        if !path.exists() {
            return Ok(BackupVerifyResult {
                valid: false,
                sha256_match: false,
                integrity_check_pass: false,
                error: Some(format!("file not found: {}", path.display())),
            });
        }

        // SHA-256 verification
        let sha256 = Self::sha256_file(path)?;

        // Check for sidecar .meta.json
        let meta_path = path.with_extension("meta.json");
        // Also try the pattern: same stem but .meta.json
        let stem = path.file_stem().and_then(|s| s.to_str()).unwrap_or("");
        let alt_meta_path = path.parent().unwrap_or(std::path::Path::new("."))
            .join(format!("{stem}.meta.json"));

        let sha256_match = if let Some(meta_file) = [meta_path, alt_meta_path]
            .iter()
            .find(|p| p.exists())
        {
            if let Ok(meta_str) = std::fs::read_to_string(meta_file) {
                if let Ok(meta) = serde_json::from_str::<serde_json::Value>(&meta_str) {
                    meta["sha256"].as_str() == Some(sha256.as_str())
                } else {
                    false
                }
            } else {
                false
            }
        } else {
            false // No sidecar found — can't verify checksum match
        };

        // SQLite integrity check — open backup and run PRAGMA
        let integrity_check_pass = match rusqlite::Connection::open(path) {
            Ok(conn) => match conn.query_row("PRAGMA integrity_check", [], |row| {
                row.get::<_, String>(0)
            }) {
                Ok(result) => result == "ok",
                Err(_) => false,
            },
            Err(_) => false,
        };

        let valid = integrity_check_pass; // valid = SQLite says it's clean
        let error = if !valid {
            Some("SQLite integrity check failed".to_string())
        } else {
            None
        };

        Ok(BackupVerifyResult {
            valid,
            sha256_match,
            integrity_check_pass,
            error,
        })
    }
```

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-04-01 | Tampering | Backup .db file | accept | Backup files sit in user's home dir — same trust boundary as main DB. User responsible for filesystem permissions. SHA-256 sidecar enables detection. |
| T-04-02 | Information Disclosure | Backup .db contents | accept | Backup is unencrypted copy of DB. If user has DB encryption enabled via Encrypt command, backup captures plaintext. Documented in REQUIREMENTS.md out-of-scope. |
| T-04-03 | Spoofing | verify-backup meta.json | mitigate | verify-backup checks SHA-256 of .db against .meta.json. If meta.json is tampered, sha256_match=false signals mismatch. Integrity check via PRAGMA provides independent validation. |
| T-04-04 | Denial of Service | Disk full during backup | accept | run_to_completion() will return SQLite error on I/O failure. Error propagated as EngramError::Database. Standard disk-full handling. |
</threat_model>
